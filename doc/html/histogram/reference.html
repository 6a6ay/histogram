Header <boost/histogram.hpp>

Includes all standard headers of the Boost.histogram library.

The library consists of two histogram implementations static_histogram and dynamic_histogram which share a common interface. The first is faster, but lacks run-time polymorphism, the second implements the opposite trade-off. Several axis types are included, which implement different binning algorithms. The axis types are passed in the constructor of the histogram to configure its binning.

Header <boost/histogram/axis.hpp>

namespace boost {
  namespace histogram {
    template<bool UOFlow> class axis_base;

    template<> class axis_base<false>;
    template<> class axis_base<true>;

    template<typename Axis> class axis_iterator;

    template<typename Value> struct bin;

    template<> struct bin<const std::string &>;

    class category_axis;
    template<typename RealType = double> class circular_axis;
    class integer_axis;

    template<typename Value> struct real_bin;

    template<typename RealType = double> class regular_axis;
    template<typename RealType = double> class variable_axis;

    typedef typename std::conditional< std::is_floating_point< Value >::value, real_bin< Value >, bin< Value > >::type axis_bin;
    typedef mpl::vector< regular_axis< double >, regular_axis< float >, circular_axis< double >, circular_axis< float >, variable_axis< double >, variable_axis< float >, integer_axis, category_axis >::type default_axes;
  }
}

Header <boost/histogram/serialization.hpp>

Defines the serialization functions, to use with boost.serialize. 

namespace boost {
  namespace histogram {
    template<typename Archive, typename Container> 
      void serialize(Archive & ar, container_storage< Container > & store, 
                     unsigned);
    template<typename Archive> 
      void serialize(Archive & ar, axis_base< false > & base, unsigned);
    template<typename Archive> 
      void serialize(Archive & ar, axis_base< true > & base, unsigned);
    template<typename Archive, typename RealType> 
      void serialize(Archive & ar, regular_axis< RealType > & axis, unsigned);
    template<typename Archive, typename RealType> 
      void serialize(Archive & ar, circular_axis< RealType > & axis, unsigned);
    template<typename Archive, typename RealType> 
      void serialize(Archive & ar, variable_axis< RealType > & axis, unsigned);
    template<typename Archive> 
      void serialize(Archive & ar, integer_axis & axis, unsigned);
    template<typename Archive> 
      void serialize(Archive & ar, category_axis & axis, unsigned);
